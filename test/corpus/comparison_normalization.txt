        a = a + 1
    Loop

    Do Until obj Is Nothing
        Set obj = obj.Parent
    Loop

    While i < 10
        i = i + 1
    Wend
End Sub

    If -foo < 0 Then Debug.Print "negative"
    If Not value Is Nothing Then Debug.Print "present"

    Select Case value
        Case Is >= 10
            Debug.Print "big"
        Case 1 To 5
            Debug.Print "small"
    End Select

    x = y
    Set obj = otherObj
End Sub
